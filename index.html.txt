<script>
  // ---------- Basic DOM references ----------
  const modal = document.getElementById('modal');
  const modalBackdrop = document.getElementById('modalBackdrop');
  const continueBtn = document.getElementById('continueBtn');
  const closeModalBtn = document.getElementById('closeModalBtn');
  const navHome = document.getElementById('nav-home');
  const navAbout = document.getElementById('nav-about');
  const navContact = document.getElementById('nav-contact');
  const aboutPanel = document.getElementById('aboutPanel');
  const contactPanel = document.getElementById('contactPanel');
  const scrollTournamentBtn = document.getElementById('scrollTournament');
  const tournamentSection = document.getElementById('tournament');

  // ---------- Modal open/close ----------
  continueBtn.addEventListener('click', () => {
    modal.classList.add('visible');
    modal.setAttribute('aria-hidden','false');
  });
  modalBackdrop.addEventListener('click', closeModal);
  closeModalBtn.addEventListener('click', closeModal);
  function closeModal(){
    modal.classList.remove('visible');
    modal.setAttribute('aria-hidden','true');
  }
  document.addEventListener('keydown', (e) => { if (e.key === 'Escape') closeModal(); });

  // ---------- Navigation (work & animate) ----------
  function setActiveLink(activeEl){
    [navHome, navAbout, navContact].forEach(a => a.classList.remove('active'));
    activeEl.classList.add('active');
  }

  navHome.addEventListener('click', (e) => {
    e.preventDefault();
    // close panels and modal, scroll to top (home)
    aboutPanel.classList.remove('open'); contactPanel.classList.remove('open'); closeModal();
    setActiveLink(navHome);
    window.scrollTo({top:0,behavior:'smooth'});
  });

  navAbout.addEventListener('click', (e) => {
    e.preventDefault();
    aboutPanel.classList.add('open'); contactPanel.classList.remove('open'); closeModal();
    setActiveLink(navAbout);
  });

  navContact.addEventListener('click', (e) => {
    e.preventDefault();
    contactPanel.classList.add('open'); aboutPanel.classList.remove('open'); closeModal();
    setActiveLink(navContact);
  });

  // Close panels with Escape or click outside (keyboard)
  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape') { aboutPanel.classList.remove('open'); contactPanel.classList.remove('open'); }
  });

  // ---------- Tournament quick scroll ----------
  scrollTournamentBtn.addEventListener('click', ()=> {
    tournamentSection.scrollIntoView({behavior:'smooth',block:'center'});
  });

  // make QR image clickable to open in new tab
  document.querySelectorAll('.qr-wrap img').forEach(img => {
    img.style.cursor = 'pointer';
    img.addEventListener('click', ()=> window.open(img.src, '_blank'));
  });

  // Accessibility: close panels when clicking outside them
  document.addEventListener('click', (e) => {
    const panels = [aboutPanel, contactPanel];
    // if click inside panel, ignore
    if (panels.some(p => p.contains(e.target))) return;
    // if click nav links, ignore because nav handlers manage it
    if (e.target.closest('nav')) return;
    // else close panels
    panels.forEach(p => p.classList.remove('open'));
    setActiveLink(navHome);
  });
</script>
